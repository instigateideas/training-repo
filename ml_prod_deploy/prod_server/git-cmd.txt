# change directory to .ssh
cd ./.ssh

# Generate the ssh key
ssh-keygen -t ed25519 -C "example@email.com"

# you will be prompted a name enter a name eg. aws-key
# then you will be prompted a passphrase: enter a 4 digit key eg 7070 (optional else leave blank by pressing enter)
# then re-enter to confirm and press enter. 
# now the public (.pub) and private key will be created in the .ssh folder

# Now cat the .pub file to update in the Github
cat aws-key.pub

# now copy the key and paste in the github
# goto account settings in github --> Select SSH & GPG Keys
# click on the add new key
# enter the title
# paste the copied data from cat aws-key.pub
# click on the aws-key

# Start the ssh-agent
eval "$(ssh-agent -s)"

# Add the private key to the ssh-agent
ssh-add ~/.ssh/aws-key
